  • 
                        '
                        public static double GetThresholdMedian(double [] array, double Threshold)
                        {
                            double aux = 0.0;
                            int count = 0;


                            for(int i = 0; i < array.Length; i++)
                            {
                                if(array[i] > Threshold)
                                {
                                    aux += array[i];
                                    count ++;
                                }
                            }
                            return aux / count;
                        }
