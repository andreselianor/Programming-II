  • FUNCIÓN: devuelve una serie de números. 
    
            '
            public static string Concat(int number)
            {
                string result = "0";
                for(int i = 0; i < number; i++)
                {
                    result += "," + i;
                }
                return result;
            }
            '
  • FUNCIÓN: devuelve una serie de números.

            public static string Concat(int number)
            {
                string result = "0";
                int multiplicador = 1;
                for(int i = 0; i < number; i++)
                {
                    multiplicador *= 2;
                    result += "," + multiplicador;
                }
                return result;
            }
